#include <bits/stdc++.h>

## B. [20联赛集训day7]序列 

区间加就差分，对操作也差分。  

无非就是要知道某个操作做了多少次，倒着差分就可以了。

```cpp
#include <bits/stdc++.h>
using namespace std;

const int N = 1e5 + 4, MOD = 1e9 + 7;
int n, q, a[N], b[N];

struct data {
	int op, l, r;
} now[N];

int main() {
    scanf("%d %d", &n, &q);
    for (int i = 1; i <= q; i++) scanf("%d %d %d", &now[i].op, &now[i].l, &now[i].r);
    b[q + 1] = 1;
    for (int i = q; i >= 1; i--) {
    	b[i] = (b[i + 1] + b[i]) % MOD;
        if (now[i].op == 1) {
        	a[now[i].l] = (a[now[i].l] + b[i]) % MOD;
        	a[now[i].r + 1] = ((a[now[i].r + 1] - b[i]) % MOD + MOD) % MOD;
        } else {
        	b[now[i].r] = (b[now[i].r] + b[i]) % MOD;
        	b[now[i].l - 1] = ((b[now[i].l - 1] - b[i]) % MOD + MOD) % MOD;
        }
    }
    for (int i = 1; i <= n; i++) {
    	a[i] = (a[i - 1] + a[i]) % MOD;
    	printf("%d ", a[i]);
    }
    return putchar('\n'), 0;
}
```
