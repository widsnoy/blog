建图之后，对于每个点遍历一次。  
只有被经过$n$次的是被除它所有人欢迎的。  

~~话说暴搜怎么过了？？？~~  

```cpp
#include <bits/stdc++.h>
using namespace std;

const int N = 1e4 + 5;
int n, m, vis[N], s[N];
vector<int> e[N];

void dfs(int u) {
    vis[u] = 1;
	s[u]++;
	for (auto v : e[u]) {
		if (!vis[v]) dfs(v);
	}
}

int main() {
    scanf("%d %d", &n, &m);
	for (int i = 1; i <= m; i++) {
		int a, b;
		scanf("%d %d", &a, &b);
		e[a].push-back(b);
	}
	for (int i = 1; i <= n; i++) {
		memset(vis, 0, sizeof vis);
		dfs(i);
	}
	int res = 0;
	for (int i = 1; i <= n; i++) res += (s[i] == n);
	printf("%d\n", res);
	return 0;
}

```
